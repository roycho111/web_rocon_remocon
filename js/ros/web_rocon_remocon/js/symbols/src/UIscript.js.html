 41 /*
 42     * Receives following ROS callbacks and throw exception for error
 43     *    * 'error'
 44     *    * 'connection'
 45     *    * 'close' 
 46     *
 47     * Updates connectionInfo text
 48     *
 49 */
 50 function settingROSCallbacks()
 51 {
 52   ros.on('error', function(error) {
 53         // throw exception for error
 54         console.log('Connection refused. Is the master running?');
 55         alert('Connection refused. Is the master running?');
 56 
 57         $("#connectBtn").show();
 58         $("#disconnectBtn").hide();
 59         
 60         InitList();
 61     });
 62 
 63     ros.on('connection', function() {
 64         console.log('Connection made!');
 65 
 66         $("#connectBtn").hide();
 67         $("#disconnectBtn").show();
 68         
 69         InitList();
 70         DisplayMasterInfo();
 71         GetRoles();
 72         MasterInfoMode();
 73     });
 74 
 75     ros.on('close', function() {
 76         console.log('Connection closed.');
 77 
 78         $("#connectBtn").show();
 79         $("#disconnectBtn").hide();
 80         
 81         InitList();
 82     });
 83 }

 85 /**
 86     * Read Cookies
 87     *
 88 */
 89 function ReadCookies()
 90 {
 91     $.cookie.defaults = { path: '/', expires: 365 };
 92 
 93     ncookieCnt = $.cookie("cookieCount");
 94 
 95     // init cookie count
 96     if (ncookieCnt == null || ncookieCnt < 0) {
 97         $.cookie("cookieCount", 0);
 98         ncookieCnt = 0;
 99     }
100 
101     // read cookie and add to url list
102     for (var i = 0; i < ncookieCnt; i++) {
103         $("#urlList").append(new Option($.cookie("cookie_url" + i)));
104         $("#urlList option:last").attr("cookieNum", i);
105     }
106 }

108 /**
109     * Event function when 'Connect' button clicked
110     *
111     *   1. get the url, hostname value out of selected url list
112     *   2. extract the url
113     *   3. connect to rosbridge
114     *
115 */
116 function Connect()
117 {
118     $("#connectBtn").click(function () {
119         var url = $("#urlList option:selected").val();
120         
121         if (url == "(Please add URL)") {
122             return;
123         }
124     
125         gUrl = url;
126 
127         // extract the exact url
128         var newurl;
129         newurl = url.replace("ws://", "");
130 
131         for (var i = 0; i < newurl.length; i++) {
132             newurl = newurl.replace("/", "");
133             newurl = newurl.replace(" ", "");
134         }
135         
136         // set default port
137         if (newurl.search(":") < 0) {
138             newurl += ":9090";
139         }
140 
141         try {
142             ros.connect('ws://' + newurl);
143             $(this).text = "Disconnect";
144         } catch(e) {
145             console.log(e.message);
146             alert(e.message);
147         }
148     });
149 }

164 /**
165     * Event function when 'Add Url' button clicked
166     *
167     *   1. get the value of currently typed url and hostname

169     *
170 */
171 function AddUrl()
172 {
173     $("#addurl_addBtn").click(function () {
174         var url = $("#typeURL").val();
175 
176         // set default string
177         if (url == "" || url == "ws://") {
178             url = "ws://localhost:9090";
179         }
180 
181         // add url
182         $("#urlList").append(new Option(url));
183         $("#urlList option:last").attr("selected", "selected");
184         $("#urlList option:last").attr("cookieNum", ncookieCnt);
185 
186         // add cookie
187         $.cookie("cookie_url" + ncookieCnt, url);
188         $.cookie("cookieCount", ++ncookieCnt);
189     });
190 }

192 /**
193     * Event function when 'Minus' button clicked
194     *
195 */
196 function DeleteUrl()
197 {
198     $("#urldeleteBtn").click(function () {
199         if ($("#urlList option:selected").val() != "(Please add URL)") {
200         
201             // delete cookie
202             var cookieNum = $("#urlList option:selected").attr("cookieNum");
203             $.cookie("cookie_url" + cookieNum, null);
204             
205             if (ncookieCnt > 0) {
206                 $.cookie("cookieCount", --ncookieCnt);
207             }
208             
209             $("#urlList option:selected").remove();
210 
211             var listCnt = $("#urlList option").length;
212             var cnt = 0;
213 
214             // rearrange cookies
215             // not including the first disabled option
216             for (var i = 1; i < listCnt; i++) {
217                 var url = $("#urlList option:eq(" + i + ")").val();
218 
219                 $("#urlList option:eq(" + i + ")").attr("cookieNum", cnt);
220                 $.cookie("cookie_url" + cnt, url);
221                 cnt++;
222             }
223         }
224     });
225 }

312 /**
313     * Classify the interaction whether it's (web_url) or (web_app)
314     *
315     *
316     *   @param : interaction - interactions[]
317 */
318 function ClassifyInteraction(interaction)
319 {
320     var newurl;
321     var url = interaction.name;
322 
323     if (url.search("web_url") >= 0) {
324         newurl = url.substring(8, url.length - 1);
325     }
326     else if (url.search("web_app") >= 0) {
327         var tempurl = url.substring(8, url.length - 1);
328         newurl = PrepareWebappUrl(interaction, tempurl);
329     }
330     else {
331         newurl = null;
332     }
333 
334     return newurl;
335 }

337 /**
338     * Url synthesiser for sending remappings and parameters information.
339     *
340     *   1. convert and set the informations
341     *       - parameter (yaml string)
342     *       - remapping (rocon_std_msgs.Remapping[])
343     *   2. Package all the data in json format and dump it to one query string variable
344     *   3. Encode the url and finish constructing
345 */
346 function PrepareWebappUrl(interaction, base_url)
347 {
348     var interaction_data = {};
349     interaction_data['display_name'] = interaction.display_name;
350     interaction_data['parameters'] = jsyaml.load(interaction.parameters);
351     interaction_data['remappings'] = {};
352 
353     $.each(interaction.remappings, function(key, value) {
354         interaction_data['remappings'][value.remap_from] = value.remap_to;
355     });
356     
357     query_string_mappings = {};
358     query_string_mappings['interaction_data'] = JSON.stringify(interaction_data);
359 
360     console.log(query_string_mappings['interaction_data']);
361     
362     var url = base_url + "?interaction_data=" + encodeURIComponent(query_string_mappings['interaction_data']);
363 
364     return url;
365 }

501 /**
502     * Wrapper function for callService
503     *
504 */
505 function CallService(ros, serviceName, serviceType, request, callBack)
506 {
507     var service = new ROSLIB.Service({
508         ros : ros,
509         name : serviceName,
510         serviceType : serviceType
511     });
512 
513     // get response
514     try {
515         service.callService(request, function(result){
516             callBack(result);
517         } , function(error) {
518             alert(error);
519             console.log(error);
520         });
521     } catch (e) {
522         console.log(message);
523         alert(e.message);
524     } 
525 }

543 /**
544     * Wrapper function for subscribing topic
545     *
546 */
547 function SubscribeTopic(ros, topicName, msgType, callBack)
548 {
549     var listener = new ROSLIB.Topic({
550         ros : ros,
551         name : topicName,
552         messageType : msgType
553     });
554     
555     listener.subscribe(function(message) {
556         callBack(message);
557         listener.unsubscribe();
558     });
559 }

561 /**
562     * Get browser name
563     *
564 */
565 function GetBrowser()
566 {
567     var agt = navigator.userAgent.toLowerCase();
568     if (agt.indexOf("chrome") != -1) return 'chrome';
569     if (agt.indexOf("opera") != -1) return 'opera';
570     if (agt.indexOf("firefox") != -1) return 'firefox';
571     if (agt.indexOf("safari") != -1) return 'safari';
572     if (agt.indexOf("msie") != -1) return 'internet_explorer';
573 }
